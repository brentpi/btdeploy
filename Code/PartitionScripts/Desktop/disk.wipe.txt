# ----- DESKTOP WINDOWS 7 ------
# delete all partitions from disk
/del /all
# create “Reserved” partition
# 1MB alignment as per Microsoft specifications for Vista/7.
/cre /pri /sz:100 /for /q /V:Reserved  /LETTER:P /NTFS:Vista /align:1MB /Y
# create “System” partition
/cre /pri /sz:50000 /for /q /V:System /LETTER:Q /NTFS:Vista /align:1MB /Y
# create “Data” partition – extended/logical
/cre /ext /Y
/cre /log /for /q /V:Data /LETTER:S /NTFS:Vista /align:1MB /Y
# ----- END DESKTOP WINDOWS 7 -----
